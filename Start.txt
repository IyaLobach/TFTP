Переход в папку:
cd IdeaProjects\TFTP

Компиляция файлов:
// встать на src
javac -d bin com\suai\view\*.java com\suai\view\model\*

Запуск файлов:
// встать на src
java -classpath ./bin com.suai.model.Server

java -classpath ./bin com.suai.model.Client

// Server
cd out\artifacts\TFTP_jar
java -jar TFTP.jar

// Client
cd out\artifacts\TFTP_jar2
java -jar TFTP.jar
